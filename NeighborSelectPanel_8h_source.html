148

245  offset + (GetCellSize() - 1) * n.GetPoint(i).GetY()));
246 
247  if(buttonRect.Contains(clickPt))
248  {
249  switch(m_selectMode)

253  break;
254  case NEIGHBOR_SELECT_ONE:
