619  void Update(OurGrid& grid)
620  {
621  KeyboardUpdate(grid);
622 
623  if (m_singleStep)
624  {
625  m_keyboardPaused = false;
626  }
627 
628  m_gridPaused = m_keyboardPaused || m_mousePaused;
629  if (!m_gridPaused)
630  {
631  Super::UpdateGrid(grid);
632  if (m_singleStep)
633  {
634  m_keyboardPaused = true;
635  m_singleStep = false;
636  }
637  }
638  else
639  {
640  const s32 ONE_THOUSAND = 1000;
641  const s32 ONE_MILLION = ONE_THOUSAND*ONE_THOUSAND;
642 
643  Sleep(0,33*ONE_MILLION); // 33 ms ~= 30 fps idle
644  }
645  }

1168  while(running)

1177 
1178  case SDL_QUIT:

1181 
1182  case SDL_MOUSEBUTTONUP:
1183  mouseButtonsDown &= ~(1<<(event.button.button));

1186 
1187  case SDL_MOUSEBUTTONDOWN:
1188  mouseButtonsDown |= 1<<(event.button.button);

1190  // FALL THROUGH

1199  break;

1216  }
