206  u64 * eds = et.GetElementDataSlotsFromType(m_elementType,m_outOfSlots);
207  if (!eds)
208  {
209  continue;
210  }
211  sum += eds[m_slot];
212  if (endOfEpoch && m_resetOnRead)
213  {
214  eds[m_slot] = 0;
215  }
216  }
