00396     void RebuildControllers()

00407       u32 totalParms = parms.GetParameterCount();

00413       {
00414         ElementParameter<CC> * parm = parms.GetParameter(i);
00415         switch (parm->GetType())
00416         {
00417         case VD::S32:
00418           AddSliderController(parm);
00419           break;
00420         case VD::BOOL:
00421           AddCheckboxController(parm);
00422           break;
00423         case VD::BITS:
00424           {
00425             ElementParameterNeighborhood<CC,SITES>* epn =
00426               dynamic_cast<ElementParameterNeighborhood<CC,SITES>*>(parm);
00427             if (epn)
00428             {
00429               AddNeighborhoodController(epn);
00430             }
00431             break;
00432           }
00433           // ELSE FALL THROUGH
00434         default:
00435           FAIL(ILLEGAL_STATE);
00436         }
00437       }
00438 
00439       /* Put in the new controllers */
00440       SPoint rpt(16, 6 + ELEMENT_RENDER_SIZE * TOTAL_ROWS);
