00001 // This is the ``Mersenne Twister'' random number generator MT19937, which
00002 // generates pseudorandom integers uniformly distributed in 0..(2^32 - 1)
00003 // starting from any odd seed in 0..(2^32 - 1).  This version is a recode
00004 // by Shawn Cokus (Cokus@math.washington.edu) on March 8, 1998 of a version by
00005 // Takuji Nishimura (who had suggestions from Topher Cooper and Marc Rieffel in
00006 // July-August 1997).
00007 //
00008 // Effectiveness of the recoding (on Goedel2.math.washington.edu, a DEC Alpha
00009 // running OSF/1) using GCC -O3 as a compiler: before recoding: 51.6 sec. to
00010 // generate 300 million random numbers; after recoding: 24.0 sec. for the same
00011 // (i.e., 46.5% of original time), so speed is now about 12.5 million random
00012 // number generations per second on this machine.
00013 //
00014 // According to the URL <http://www.math.keio.ac.jp/~matumoto/emt.html>
00015 // (and paraphrasing a bit in places), the Mersenne Twister is ``designed
00016 // with consideration of the flaws of various existing generators,'' has
00017 // a period of 2^19937 - 1, gives a sequence that is 623-dimensionally
00018 // equidistributed, and ``has passed many stringent tests, including the
00019 // die-hard test of G. Marsaglia and the load test of P. Hellekalek and
00020 // S. Wegenkittl.''  It is efficient in memory usage (typically using 2506
00021 // to 5012 bytes of static data, depending on data type sizes, and the code
00022 // is quite short as well).  It generates random numbers in batches of 624
00023 // at a time, so the caching and pipelining of modern systems is exploited.
00024 // It is also divide- and mod-free.

00073   uint32   initseed;    //
00074   int      left;        // can *next++ this many times before reloading
00075 
00076   inline uint32 hiBit(uint32 u) { 
00077     return u & 0x80000000U;    // mask all but highest   bit of u
00078   }
00079 
00080   inline uint32 loBit(uint32 u) { 
00081     return u & 0x00000001U;    // mask all but lowest    bit of u
00082   }
00083 
00084   inline uint32 loBits(uint32 u) { 
00085     return u & 0x7FFFFFFFU;   // mask     the highest   bit of u
00086   }
00087 
00088   inline uint32 mixBits(uint32 u, uint32 v) {
00089     return hiBit(u)|loBits(v);  // move hi bit of u to hi bit of v
00090   }

00107 inline uint32 RandMT::randomMT(void) {
00108   uint32 y;
00109 
00110   if(--left < 0)
00111     return(reloadMT());
00112 
00113   y  = *next++;
00114   y ^= (y >> 11);
00115   y ^= (y <<  7) & 0x9D2C5680U;
00116   y ^= (y << 15) & 0xEFC60000U;
