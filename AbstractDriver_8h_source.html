234  void UpdateGrid(OurGrid& grid)
235  {
236  const s32 ONE_THOUSAND = 1000;
237  const s32 ONE_MILLION = ONE_THOUSAND*ONE_THOUSAND;
238 
239  grid.Unpause(); // pausing and unpausing should be overhead!
240 
241  u32 startMS = GetTicks(); // So get the ticks after unpausing
242  if (m_ticksLastStopped != 0)
243  m_msSpentOverhead += startMS - m_ticksLastStopped;
244  else
245  m_msSpentOverhead = 0;
246 
247  Sleep(m_microsSleepPerFrame/ONE_MILLION,
248  (u64) (m_microsSleepPerFrame%ONE_MILLION)*ONE_THOUSAND);
249 
250  m_ticksLastStopped = GetTicks(); // and before pausing
251 
252  grid.Pause();
253 
254  u32 thisPeriodMS = m_ticksLastStopped - startMS;
255  m_msSpentRunning += thisPeriodMS;
256 
257  u64 totalEvents = grid.GetTotalEventsExecuted();
258  u32 totalSites = grid.GetTotalSites();
259  m_AEPS = totalEvents / totalSites;
260  m_AER = 1000 * (m_AEPS / m_msSpentRunning);
261 
262  u64 newEvents = totalEvents - m_lastTotalEvents;
263  m_lastTotalEvents = totalEvents;
264 
265  if (thisPeriodMS == 0) {
266  LOG.Warning("Zero ms in sample");
267  thisPeriodMS = 1;
268  }
269  double thisAERsample = 1000.0 * newEvents / totalSites / thisPeriodMS;
270 
271  const double BACKWARDS_AVERAGE_RATE = 0.99;
272  m_recentAER = BACKWARDS_AVERAGE_RATE * m_recentAER +
273  (1 - BACKWARDS_AVERAGE_RATE) * thisAERsample;
274 
275  m_overheadPercent = 100.0*m_msSpentOverhead/(m_msSpentRunning+m_msSpentOverhead);
276 
277  double diff = m_AEPS - m_lastFrameAEPS;
278  double err = MIN(1.0, MAX(-1.0, m_aepsPerFrame - diff));
279 
280  // Correct up to 20% of current each frame
281  m_microsSleepPerFrame = (100+20*err)*m_microsSleepPerFrame/100;
282  m_microsSleepPerFrame = MIN(100000000, MAX(1000, m_microsSleepPerFrame));
283 
284  m_lastFrameAEPS = m_AEPS;
285 
286  CheckEpochProcessing(grid);
287 
288  PostUpdate();
289  }

779  virtual void DoEpochEvents(OurGrid& grid, u32 epochs, u32 epochAEPS)
780  {
781  LOG.Debug("Epoch %d: %d AEPS", epochs, epochAEPS);
782 
783  grid.CheckCaches();
784 
785  grid.RecountAtoms();
786 
787  if (m_gridImages)
788  {
789  const char * path = GetSimDirPathTemporary("eps/%010d.ppm", epochAEPS);
790  FILE* fp = fopen(path, "w");
791  FileByteSink fbs(fp);
792  grid.WriteEPSImage(fbs);
793  fclose(fp);
794  }
795 
796  if (m_tileImages)
797  {
798  const char * path = GetSimDirPathTemporary("teps/%010d-average.ppm", epochAEPS);
799  FILE* fp = fopen(path, "w");
800  FileByteSink fbs2(fp);
801  grid.WriteEPSAverageImage(fbs2);
802  fclose(fp);
803  }
804 
805 
806 
807  if (m_autosavePerEpochs > 0 && (epochs % m_autosavePerEpochs) == 0)
808  {
809  this->AutosaveGrid(epochs);
810  }
811 
812  if (m_accelerateAfterEpochs > 0 && (epochs % m_accelerateAfterEpochs) == 0)
813  {
814  this->SetAEPSPerEpoch(this->GetAEPSPerEpoch() + m_acceleration);
815  }
816 
817  if (m_accelerateAfterEpochs > 0 &&
818  m_surgeAfterEpochs > 0 &&
819  (epochs % m_surgeAfterEpochs) == 0)
820  {
821  ++m_acceleration;
822  }
823  }
