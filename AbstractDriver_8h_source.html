00234     void UpdateGrid(OurGrid& grid)
00235     {
00236       const s32 ONE_THOUSAND = 1000;
00237       const s32 ONE_MILLION = ONE_THOUSAND*ONE_THOUSAND;
00238 
00239       grid.Unpause();  // pausing and unpausing should be overhead!
00240 
00241       u32 startMS = GetTicks();  // So get the ticks after unpausing
00242       if (m_ticksLastStopped != 0)
00243         m_msSpentOverhead += startMS - m_ticksLastStopped;
00244       else
00245         m_msSpentOverhead = 0;
00246 
00247       Sleep(m_microsSleepPerFrame/ONE_MILLION,
00248             (u64) (m_microsSleepPerFrame%ONE_MILLION)*ONE_THOUSAND);
00249 
00250       m_ticksLastStopped = GetTicks(); // and before pausing
00251 
00252       grid.Pause();
00253 
00254       u32 thisPeriodMS = m_ticksLastStopped - startMS;
00255       m_msSpentRunning += thisPeriodMS;
00256 
00257       u64 totalEvents = grid.GetTotalEventsExecuted();
00258       u32 totalSites = grid.GetTotalSites();
00259       m_AEPS = totalEvents / totalSites;
00260       m_AER = 1000 * (m_AEPS / m_msSpentRunning);
00261 
00262       u64 newEvents = totalEvents - m_lastTotalEvents;
00263       m_lastTotalEvents = totalEvents;
00264 
00265       if (thisPeriodMS == 0) {
00266         LOG.Warning("Zero ms in sample");
00267         thisPeriodMS = 1;
00268       }
00269       double thisAERsample = 1000.0 * newEvents / totalSites / thisPeriodMS;
00270 
00271       const double BACKWARDS_AVERAGE_RATE = 0.99;
00272       m_recentAER = BACKWARDS_AVERAGE_RATE * m_recentAER +
00273                     (1 - BACKWARDS_AVERAGE_RATE) * thisAERsample;
00274 
00275       m_overheadPercent = 100.0*m_msSpentOverhead/(m_msSpentRunning+m_msSpentOverhead);
00276 
00277       double diff = m_AEPS - m_lastFrameAEPS;
00278       double err = MIN(1.0, MAX(-1.0, m_aepsPerFrame - diff));
00279 
00280       // Correct up to 20% of current each frame
00281       m_microsSleepPerFrame = (100+20*err)*m_microsSleepPerFrame/100;
00282       m_microsSleepPerFrame = MIN(100000000, MAX(1000, m_microsSleepPerFrame));
00283 
00284       m_lastFrameAEPS = m_AEPS;
00285 
00286       CheckEpochProcessing(grid);
00287
