396  void RebuildControllers()
397  {
398  /* Remove any old controllers */
399  for(u32 i = 0; i < m_controllerCount; i++)
400  {
401  Panel::Remove(m_controllers[i]);
402  }
403  m_controllerCount = 0;
404 
405  /* Set up the new controllers, if any */
406  ElementParameters<CC> & parms = m_primaryElement->GetElementParameters();
407  u32 totalParms = parms.GetParameterCount();
408 
409  m_sliderCount = 0;
410  m_checkboxCount = 0;
411  m_neighborhoodCount = 0;
412  for(u32 i = 0; i < totalParms; i++)
413  {
414  ElementParameter<CC> * parm = parms.GetParameter(i);
415  switch (parm->GetType())
416  {
417  case VD::S32:
418  AddSliderController(parm);
419  break;
420  case VD::BOOL:
421  AddCheckboxController(parm);
422  break;
423  case VD::BITS:
424  {
425  ElementParameterNeighborhood<CC,SITES>* epn =
426  dynamic_cast<ElementParameterNeighborhood<CC,SITES>*>(parm);
427  if (epn)
428  {
429  AddNeighborhoodController(epn);
430  }
431  break;
432  }
433  // ELSE FALL THROUGH
434  default:
435  FAIL(ILLEGAL_STATE);
436  }
437  }
438 
439  /* Put in the new controllers */
440  SPoint rpt(16, 6 + ELEMENT_RENDER_SIZE * TOTAL_ROWS);
441  for(u32 i = 0; i < m_controllerCount; i++)
442  {
443  ParameterController<CC> * c = m_controllers[i];
444  c->SetRenderPoint(rpt);
445  Panel::Insert(c, NULL);
446  UPoint desired = c->GetDesiredSize();
447  rpt.SetY(rpt.GetY() + desired.GetY());
448  }
449 
450  Panel::SetDimensions(6 + ELEMENT_RENDER_SIZE * ELEMENTS_PER_ROW + BORDER_PADDING,
451  rpt.GetY() + BORDER_PADDING);
452  Panel::SetDesiredSize(6 + ELEMENT_RENDER_SIZE * ELEMENTS_PER_ROW + BORDER_PADDING,
453  rpt.GetY() + BORDER_PADDING);
454 
455  //debug: this->Print(STDOUT);
456 
457  // Tell ourselfs to resize (our cheeso repacking)
458  HandleResize(this->m_parent->GetDimensions());
459 
460  }

478  {
479  m_toolButtons[i].SetParent(this);
480  m_toolButtons[i].Panel::SetRenderPoint(SPoint(16 + i * 32, 3));
481  m_toolButtons[i].SetToolIcon(AssetManager::Get(assets[i]));
482  Panel::Insert(m_toolButtons + i, NULL);
483  }
484 
485  for(u32 i = 0; i < ELEMENT_BOX_SIZE; i++)
486  {
487  u32 x, y;

489  x = 3 + ELEMENT_RENDER_SIZE * (i % ELEMENTS_PER_ROW);
490  y = 3 + ELEMENT_RENDER_SIZE * (3 + (i / ELEMENTS_PER_ROW));
491 
492  /* These will render correctly because ElementButtons render
493  grey when SetElement is fed NULL. */
494  m_elementButtons[i].SetElement(m_heldElements[i]);
495  m_elementButtons[i].SetParent(this);
496  m_elementButtons[i].Panel::SetRenderPoint(SPoint(x, y));
497 
498  Panel::Insert(m_elementButtons + i, NULL);
499  }
500 
501  m_primaryElement = m_heldElements[0];
502  m_secondaryElement = m_heldElements[1];
503 
504  Panel::SetDimensions(6 + ELEMENT_RENDER_SIZE * ELEMENTS_PER_ROW,
505  6 + ELEMENT_RENDER_SIZE * TOTAL_ROWS);
506  Panel::SetDesiredSize(6 + ELEMENT_RENDER_SIZE * ELEMENTS_PER_ROW,
507  6 + ELEMENT_RENDER_SIZE * TOTAL_ROWS);
508 
509  /* Set pencil tool to default */
