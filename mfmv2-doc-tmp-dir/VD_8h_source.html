95  static s32 GetMin(u32 type, u32 bits)
96  {
97  switch (type)
98  {
99  case U32: return 0;
100  case S32: return bits==0 ? 0 : -MakeMask(bits-1) - 1;

108  static s32 GetMax(u32 type, u32 bits)
109  {
110  switch (type)
