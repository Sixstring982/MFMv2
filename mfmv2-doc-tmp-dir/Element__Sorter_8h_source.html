157  virtual void Behavior(EventWindow<CC>& window) const
158  {
159  Random & random = window.GetRandom();
160  SPoint reproducePt;
161  T self = window.GetCenterAtom();
162  if(this->FillPointWithType(window, reproducePt,
163  Element<CC>::VNNeighbors, 4,
164  Dirs::SOUTHEAST, Element_Res<CC>::TYPE()))
165  {
166  window.SetRelativeAtom(reproducePt, self);
167  }
168 
169  SPoint seData, neData, swEmpty, nwEmpty, srcPt, dstPt;
170  bool movingUp = random.CreateBool();
171 
172  for(s32 i = 0; i < 2; i++)
173  {
174  if(movingUp &&
175  FillAvailableSubwindowPoint(window, seData,
176  Dirs::SOUTHEAST,
177  Element_Data<CC>::THE_INSTANCE.GetType()) &&
178  FillAvailableSubwindowPoint(window, nwEmpty,
179  Dirs::NORTHWEST,
180  Element_Empty<CC>::THE_INSTANCE.GetType()))
181  {
182  srcPt = seData;
183  dstPt = nwEmpty;
184  }
185  else if(!movingUp &&
186  FillAvailableSubwindowPoint(window, neData,
187  Dirs::NORTHEAST,
188  Element_Data<CC>::THE_INSTANCE.GetType()) &&
189  FillAvailableSubwindowPoint(window, swEmpty,
190  Dirs::SOUTHWEST,
191  Element_Empty<CC>::THE_INSTANCE.GetType()))
192  {
193  srcPt = neData;
194  dstPt = swEmpty;
195  }
196  else
197  {
198  movingUp = !movingUp;
199  continue;
200  }
201 
202  u32 threshold = GetThreshold(window.GetCenterAtom(),0);
203  u32 datum = Element_Data<CC>::THE_INSTANCE.GetDatum(window.GetRelativeAtom(srcPt),0);
204  u32 cmp = (movingUp && (datum < threshold)) || (!movingUp && (datum > threshold));
205  if(cmp)
206  {
207  SetThreshold(self,datum);
208  window.SetCenterAtom(self);
209  window.SwapAtoms(srcPt, dstPt);
210  break;
211  }
212  }
213  this->Diffuse(window);
214  }
